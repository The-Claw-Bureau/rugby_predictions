            es.event_id,

            vs.venue_id AS venue_internal_id,
            s2.name AS external_type,
            c1.hemisphere,

             LEFT JOIN team_source_comp ts1 ON es.source_id = ts1.source_id AND es.home_team_external_id::text = ts1.external_id::text AND es.competition_external_id::text = ts1.competition_external_id::text
             LEFT JOIN team_source_comp ts2 ON es.source_id = ts2.source_id AND es.away_team_external_id::text = ts2.external_id::text AND es.competition_external_id::text = ts2.competition_external_id::text

             LEFT JOIN team t2_1 ON ts2.team_id = t2_1.id
             LEFT JOIN competition_source cs ON es.source_id = cs.source_id AND es.competition_external_id::text = cs.external_id::text
             LEFT JOIN competition c1 ON cs.competition_id = c1.id
             LEFT JOIN venue_source vs ON es.source_id = vs.source_id AND es.venue_external_id::text = vs.external_id::text
             LEFT JOIN source s2 ON es.source_id = s2.id
             LEFT JOIN source_priority oop ON s2.id = oop.source_id
